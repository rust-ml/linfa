14/// Calculate the Pearson's Correlation Coefficient (or bivariate correlation)
15///
16/// The PCC describes the linear correlation between two variables. It is the covariance divided by
17/// the product of the standard deviations, therefore essentially a normalised measurement of the
18/// covariance and in range (-1, 1). A negative coefficient indicates a negative correlation
19/// between both variables.
20fn pearson_correlation<F: Float, D: Data<Elem = F>>(data: &ArrayBase<D, Ix2>) -> Array1<F> {
21    // number of obserations and features
22    let nobservations = data.nrows();
23    let nfeatures = data.ncols();
24
25    // center distribution by subtracting mean
26    let mean = data.mean_axis(Axis(0)).unwrap();
27    //let std_deviation = mean.clone();
28    let denoised = data - &mean.insert_axis(Axis(1)).t();
29
30    // calculate the covariance matrix
31    let covariance = denoised.t().dot(&denoised) / F::cast(nobservations - 1);
32    // calculate the standard deviation vector
33    let std_deviation = denoised.var_axis(Axis(0), F::one()).mapv(|x| x.sqrt());
34
35    // we will only save the upper triangular matrix as the diagonal is one and
36    // the lower triangular is a mirror of the upper triangular part
37    let mut pearson_coeffs = Array1::zeros(nfeatures * (nfeatures - 1) / 2);
38
39    let mut k = 0;
40    for i in 0..(nfeatures - 1) {
41        for j in (i + 1)..nfeatures {
42            // calculate pearson correlation coefficients by normalizing the covariance matrix
43            pearson_coeffs[k] = covariance[(i, j)] / std_deviation[i] / std_deviation[j];
44
45            k += 1;
46        }
47    }
48
49    pearson_coeffs
50}

52/// Evidence of non-correlation with re-sampling test
53///
54/// The p-value supports or reject the null hypthesis that two variables are not correlated. A
55/// small p-value indicates a strong evidence that two variables are correlated.
56fn p_values<F: Float, D: Data<Elem = F>>(
57    data: &ArrayBase<D, Ix2>,
58    ground: &Array1<F>,
59    num_iter: usize,
60) -> Array1<F> {
61    // transpose element matrix such that we can shuffle columns
62    let (n, m) = (data.ncols(), data.nrows());
63    let mut flattened = Vec::with_capacity(n * m);
64    for i in 0..m {
65        for j in 0..n {
66            flattened.push(data[(i, j)]);
67        }
68    }
69
70    let mut p_values = Array1::zeros(n * (n - 1) / 2);
71    let mut rng = SmallRng::from_entropy();
72
73    // calculate p-values by shuffling features `num_iter` times
74    for _ in 0..num_iter {
75        // shuffle all corresponding features
76        for j in 0..n {
77            flattened[j * m..(j + 1) * m].shuffle(&mut rng);
78        }
79
80        // create an ndarray and calculate the PCC for this distribution
81        let arr_view = ArrayView2::from_shape((m, n), &flattened).unwrap();
82        let correlation = pearson_correlation(&arr_view.t());
83
84        // count the number of times that the re-shuffled distribution has a larger PCC than the
85        // original distribution
86        let greater = ground
87            .iter()
88            .zip(correlation.iter())
89            .map(|(a, b)| {
90                if a.abs() < b.abs() {
91                    F::one()
92                } else {
93                    F::zero()
94                }
95            })
96            .collect::<Array1<_>>();
97
98        p_values += &greater;
99    }
100
101    // divide by the number of iterations to re-scale range
102    p_values / F::cast(num_iter)
103}

105/// Pearson Correlation Coefficients (or Bivariate Coefficients)
106///
107/// The PCCs indicate the linear correlation between variables. This type also supports printing
108/// the PCC as an upper triangle matrix together with the feature names.
109#[derive(Debug, Clone, PartialEq, Eq)]
110pub struct PearsonCorrelation<F> {
111    pearson_coeffs: Array1<F>,
112    p_values: Array1<F>,
113    feature_names: Vec<String>,
114}
115
116impl<F: Float> PearsonCorrelation<F> {
117    /// Calculate the Pearson Correlation Coefficients and optionally p-values from dataset
118    ///
119    /// The PCC describes the linear correlation between two variables. It is the covariance divided by
120    /// the product of the standard deviations, therefore essentially a normalised measurement of the
121    /// covariance and in range (-1, 1). A negative coefficient indicates a negative correlation
122    /// between both variables.
123    ///
124    /// The p-value supports or reject the null hypthesis that two variables are not correlated. A
125    /// small p-value indicates a strong evidence that two variables are correlated.
126    ///
127    /// # Parameters
128    ///
129    /// * `dataset`: Data for the correlation analysis
130    /// * `num_iter`: optionally number of iterations of the p-value test, if none then no p-value
131    ///   are calculate
132    ///
133    /// # Example
134    ///
135    /// ```
136    /// let corr = linfa_datasets::diabetes()
137    ///     .pearson_correlation_with_p_value(100);
138    ///
139    /// println!("{}", corr);
140    /// ```
141    ///
142    /// The output looks like this (the p-value is in brackets behind the PCC):
143    ///
144    /// ```ignore
145    /// age                        +0.17 (0.61) +0.18 (0.62) +0.33 (0.34) +0.26 (0.47) +0.22 (0.54) -0.07 (0.83) +0.20 (0.60) +0.27 (0.54) +0.30 (0.41)
146    /// sex                                     +0.09 (0.74) +0.24 (0.59) +0.04 (0.91) +0.14 (0.74) -0.38 (0.28) +0.33 (0.30) +0.15 (0.74) +0.21 (0.58)
147    /// body mass index                                      +0.39 (0.20) +0.25 (0.45) +0.26 (0.51) -0.37 (0.31) +0.41 (0.24) +0.45 (0.21) +0.39 (0.21)
148    /// blood pressure                                                    +0.24 (0.54) +0.19 (0.56) -0.18 (0.61) +0.26 (0.45) +0.39 (0.20) +0.39 (0.16)
149    /// t-cells                                                                        +0.90 (0.00) +0.05 (0.89) +0.54 (0.05) +0.52 (0.10) +0.33 (0.37)
150    /// low-density lipoproteins                                                                    -0.20 (0.53) +0.66 (0.04) +0.32 (0.42) +0.29 (0.42)
151    /// high-density lipoproteins                                                                                -0.74 (0.02) -0.40 (0.21) -0.27 (0.42)
152    /// thyroid stimulating hormone                                                                                           +0.62 (0.04) +0.42 (0.21)
153    /// lamotrigine                                                                                                                        +0.47 (0.14)
154    /// blood sugar level
155    /// ```
156    pub fn from_dataset<D: Data<Elem = F>, T>(
157        dataset: &DatasetBase<ArrayBase<D, Ix2>, T>,
158        num_iter: Option<usize>,
159    ) -> Self {
160        // calculate pearson coefficients
161        let pearson_coeffs = pearson_correlation(dataset.records());
162
163        // calculate p values
164        let p_values = match num_iter {
165            Some(num_iter) => p_values(dataset.records(), &pearson_coeffs, num_iter),
166            None => Array1::zeros(0),
167        };
168
169        PearsonCorrelation {
170            pearson_coeffs,
171            p_values,
172            feature_names: dataset.feature_names().to_vec(),
173        }
174    }
175
176    /// Return the Pearson's Correlation Coefficients
177    ///
178    /// The coefficients are describing the linear correlation, normalized in range (-1, 1) between
179    /// two variables. Because the correlation is commutative and PCC to the same variable is
180    /// always perfectly correlated (i.e. 1), this function only returns the upper triangular
181    /// matrix with (n-1)*n/2 elements.
182    pub fn get_coeffs(&self) -> &Array1<F> {
183        &self.pearson_coeffs
184    }
185
186    /// Return the p values supporting the null-hypothesis
187    ///
188    /// This implementation estimates the p value with the permutation test. As null-hypothesis
189    /// the non-correlation between two variables is chosen such that the smaller the p-value the
190    /// stronger we can reject the null-hypothesis and conclude that they are linearily correlated.
191    pub fn get_p_values(&self) -> Option<&Array1<F>> {
192        if self.p_values.is_empty() {
193            None
194        } else {
195            Some(&self.p_values)
196        }
197    }
198}

200impl<F: Float, D: Data<Elem = F>, T> DatasetBase<ArrayBase<D, Ix2>, T> {
201    /// Calculate the Pearson Correlation Coefficients from a dataset
202    ///
203    /// The PCC describes the linear correlation between two variables. It is the covariance divided by
204    /// the product of the standard deviations, therefore essentially a normalised measurement of the
205    /// covariance and in range (-1, 1). A negative coefficient indicates a negative correlation
206    /// between both variables.
207    ///
208    /// # Example
209    ///
210    /// ```
211    /// let corr = linfa_datasets::diabetes()
212    ///     .pearson_correlation();
213    ///
214    /// println!("{}", corr);
215    /// ```
216    ///
217    pub fn pearson_correlation(&self) -> PearsonCorrelation<F> {
218        PearsonCorrelation::from_dataset(self, None)
219    }
220
221    /// Calculate the Pearson Correlation Coefficients and p-values from the dataset
222    ///
223    /// The PCC describes the linear correlation between two variables. It is the covariance divided by
224    /// the product of the standard deviations, therefore essentially a normalised measurement of the
225    /// covariance and in range (-1, 1). A negative coefficient indicates a negative correlation
226    /// between both variables.
227    ///
228    /// The p-value supports or reject the null hypthesis that two variables are not correlated.
229    /// The smaller the p-value the stronger is the evidence that two variables are correlated. A
230    /// typical threshold is p < 0.05.
231    ///
232    /// # Parameters
233    ///
234    /// * `num_iter`: number of iterations of the permutation test to estimate the p-value
235    ///
236    /// # Example
237    ///
238    /// ```
239    /// let corr = linfa_datasets::diabetes()
240    ///     .pearson_correlation_with_p_value(100);
241    ///
242    /// println!("{}", corr);
243    /// ```
244    ///
245    pub fn pearson_correlation_with_p_value(&self, num_iter: usize) -> PearsonCorrelation<F> {
246        PearsonCorrelation::from_dataset(self, Some(num_iter))
247    }
248}

250/// Display the Pearson's Correlation Coefficients as upper triangular matrix
251///
252/// This function prints the feature names for each row, the corresponding PCCs and optionally the
253/// p-values in brackets after the PCCs.
254impl<F: Float> fmt::Display for PearsonCorrelation<F> {
255    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
256        let n = self.feature_names.len();
257        let longest = self.feature_names.iter().map(|x| x.len()).max().unwrap();
258
259        let mut k = 0;
260        for i in 0..(n - 1) {
261            write!(f, "{}", self.feature_names[i])?;
262            for _ in 0..longest - self.feature_names[i].len() {
263                write!(f, " ")?;
264            }
265
266            for _ in 0..i {
267                write!(f, "             ")?;
268            }
269
270            for _ in (i + 1)..n {
271                if !self.p_values.is_empty() {
272                    write!(
273                        f,
274                        "{:+.2} ({:.2}) ",
275                        self.pearson_coeffs[k], self.p_values[k]
276                    )?;
277                } else {
278                    write!(f, "{:.2} ", self.pearson_coeffs[k])?;
279                }
280
281                k += 1;
282            }
283            writeln!(f,)?;
284        }
285        writeln!(f, "{}", self.feature_names[n - 1])?;
286
287        Ok(())
288    }
289}

291#[cfg(test)]
292mod tests {
293    use crate::DatasetBase;
294    use ndarray::{concatenate, Array, Axis};
295    use ndarray_rand::{rand_distr::Uniform, RandomExt};
296    use rand::{rngs::SmallRng, SeedableRng};
297
298    #[test]
299    fn uniform_random() {
300        // create random number generator and random matrix with uniform distribution
301        let mut rng = SmallRng::seed_from_u64(42);
302        let data = Array::random_using((1000, 4), Uniform::new(-1., 1.), &mut rng);
303
304        // calculate PCCs and test that they are indeed near zero
305        let pcc = DatasetBase::from(data).pearson_correlation();
306        assert!(pcc.get_coeffs().mapv(|x: f32| x.abs()).sum() < 5e-2 * 6.0);
307    }
308
309    #[test]
310    fn perfectly_correlated() {
311        let mut rng = SmallRng::seed_from_u64(42);
312        let v = Array::random_using((4, 1), Uniform::new(0., 1.), &mut rng);
313
314        // project feature with matrix
315        let data = Array::random_using((1000, 1), Uniform::new(-1., 1.), &mut rng);
316        let data_proj = data.dot(&v.t());
317
318        let corr = DatasetBase::from(concatenate![Axis(1), data, data_proj])
319            .pearson_correlation_with_p_value(100);
320
321        assert!(corr.get_coeffs().mapv(|x| 1. - x).sum() < 1e-2);
322        assert!(corr.get_p_values().unwrap().sum() < 1e-2);
323    }
