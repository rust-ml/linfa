19impl<F, D, T> Fit<ArrayBase<D, Ix2>, T, LarsError> for LarsValidParams<F>
20where
21    T: AsSingleTargets<Elem = F>,
22    D: Data<Elem = F>,
23    F: Float,
24{
25    type Object = Lars<F>;
26    /// Fit an LARS model given a feature matrix `x` and a target variable `y`.
27    ///
28    /// The feature matrix `x` must have shape `(n_samples, n_features)`
29    /// The target variable `y` must have shape `(n_samples)`
30    ///
31    /// Returns a `LARS` object which contains the fitted
32    /// parameters and can be used to `predict` values of the target variable
33    /// for new feature values.
34    fn fit(
35        &self,
36        dataset: &linfa::DatasetBase<ArrayBase<D, Ix2>, T>,
37    ) -> Result<Self::Object, LarsError> {
38        let targets = dataset.as_single_targets();
39        let (intercept, y) = compute_intercept(self.fit_intercept(), targets);
40
41        let (alphas, active, coef_path, n_iter) = lars_path(
42            &dataset.records().view(),
43            &y.view(),
44            self.n_nonzero_coefs(),
45            self.eps(),
46            self.verbose(),
47            F::zero(),
48        );
49
50        // let intercept = intercept.into_scalar();
51
52        let hyperplane = coef_path.slice(s![.., -1]).to_owned();
53
54        Ok(Lars {
55            hyperplane,
56            intercept,
57            alphas,
58            n_iter,
59            active,
60            coef_path,
61        })
62    }
63}

65/// Compute the intercept as the mean of `y` along each column and center `y` if an intercept
66/// should be used, use 0 as intercept and leave `y` unchanged otherwise.
67fn compute_intercept<F: Float>(with_intercept: bool, y: ArrayView1<F>) -> (F, Array1<F>) {
68    if with_intercept {
69        let y_mean = y
70            // Take the mean of y
71            .mean()
72            .expect("Axis 0 length of 0");
73        // Subtract y_mean from each element of y
74        let y_centered = &y - y_mean;
75        (y_mean, y_centered)
76    } else {
77        (F::zero(), y.to_owned())
78    }
79}

81/// Compute Least Angle Regression using LARS algorithm.
82/// Based on scikit-learn’s `lars_path` algorithm.
83/// https://scikit-learn.org/stable/modules/generated/sklearn.linear_model.lars_path.html#sklearn.linear_model.lars_path
84///
85/// returns alphas, active, coef_path, n_iter
86fn lars_path<F: Float>(
87    x: &ArrayView2<F>,
88    y: &ArrayView1<F>,
89    max_iter: usize,
90    eps: F,
91    verbose: bool,
92    alpha_min: F,
93) -> (Array1<F>, Vec<usize>, Array2<F>, usize) {
94    let n_samples = F::from(y.len()).unwrap();
95    let n_features = x.nfeatures();
96
97    let max_features = max_iter.min(n_features);
98
99    let mut coefs = Array2::<F>::zeros((max_features + 1, n_features));
100    let mut alphas = Array1::<F>::zeros(max_features + 1);
101
102    let mut prev_coef = Array1::<F>::zeros(n_features);
103    let mut prev_alpha = Array1::<F>::from_elem(1, F::zero());
104
105    let mut gram = x.t().dot(x);
106    let mut cov = x.t().dot(y);
107    let mut l = Array2::<F>::default((max_features, max_features));
108
109    let mut n_iter = 0;
110    let mut n_active = 0;
111
112    let mut sign_active = Array1::<F>::zeros(max_features);
113    let mut indices = Array1::<F>::from_iter((0..n_features).map(|i| F::from(i).unwrap()));
114
115    let tiny_32 = F::min_positive_value();
116    let equality_tolerance = F::epsilon();
117
118    let mut active: Vec<usize> = vec![];
119    let mut drop = false;
120
121    if verbose {
122        println!("Step\t\tAdded\t\tDropped\t\tActive set size\t\tC");
123    }
124
125    loop {
126        let c;
127        let mut c_idx = 0;
128        let mut c_ = F::zero();
129        if !cov.is_empty() {
130            c_idx = cov.abs().argmax().unwrap();
131            c_ = cov[c_idx];
132            c = c_.abs();
133        } else {
134            c = F::zero();
135        }
136
137        let mut alpha = alphas.slice(s![n_iter, NewAxis]).to_owned();
138        let mut coef = coefs.row(n_iter).to_owned();
139        if n_iter > 0 {
140            prev_alpha = alphas.slice(s![n_iter - 1, NewAxis]).to_owned();
141            prev_coef = coefs.row(n_iter - 1).to_owned();
142        }
143
144        alpha[0] = c / n_samples;
145        alphas[n_iter] = c / n_samples;
146
147        if alpha[0] <= alpha_min + equality_tolerance {
148            if (alpha[0] - alpha_min).abs() > equality_tolerance {
149                if n_iter > 0 {
150                    let ss = (prev_alpha[0] - alpha_min) / (prev_alpha[0] - alpha[0]);
151                    coef.assign(&(&prev_coef + &(&coef - &prev_coef) * ss));
152                }
153                alpha[0] = alpha_min;
154            }
155            coefs.row_mut(n_iter).assign(&coef);
156            break;
157        }
158
159        if n_iter >= max_iter || n_active >= n_features {
160            break;
161        }
162
163        if !drop {
164            sign_active[n_active] = if c_ > F::zero() {
165                F::one()
166            } else if c_ < F::zero() {
167                -F::one()
168            } else {
169                F::zero()
170            };
171
172            let m = n_active;
173            let n = c_idx + n_active;
174
175            cov.swap(c_idx, 0);
176            indices.swap(m, n);
177
178            let cov_not_shortened = cov.clone();
179            cov.remove_index(Axis(0), 0);
180
181            if m != n {
182                let n_cols = gram.ncols();
183                let (mut row_m, mut row_n) = gram.multi_slice_mut((s![m, ..], s![n, ..]));
184                for j in 0..n_cols {
185                    swap(&mut row_m[j], &mut row_n[j]);
186                }
187                let n_rows = gram.nrows();
188                let (mut col_m, mut col_n) = gram.multi_slice_mut((s![.., m], s![.., n]));
189                for j in 0..n_rows {
190                    swap(&mut col_m[j], &mut col_n[j]);
191                }
192            }
193
194            let c_diff = gram[[n_active, n_active]];
195
196            l.slice_mut(s![n_active, 0..n_active])
197                .assign(&gram.slice(s![n_active, 0..n_active]));
198
199            if n_active != 0 {
200                let l_sub = l.slice(s![..n_active, ..n_active]).to_owned().with_lapack();
201                let mut b = l
202                    .slice(s![n_active, ..n_active])
203                    .insert_axis(Axis(1))
204                    .to_owned()
205                    .with_lapack();
206
207                #[cfg(not(feature = "blas"))]
208                l_sub.solve_triangular_inplace(&mut b, UPLO::Lower).unwrap();
209
210                #[cfg(feature = "blas")]
211                l_sub
212                    .solve_triangular_inplace(UPLO::Lower, Diag::NonUnit, &mut b)
213                    .unwrap();
214
215                l.slice_mut(s![n_active, ..n_active])
216                    .assign(&(b.without_lapack().remove_axis(Axis(1))));
217            }
218
219            let row_slice = l.slice(s![n_active, 0..n_active]);
220            let v = F::from(row_slice.dot(&row_slice)).unwrap();
221
222            let diag = (c_diff - v).abs().sqrt().max(eps);
223            l[[n_active, n_active]] = diag;
224
225            if diag < F::cast(1e-7) {
226                cov.assign(&cov_not_shortened);
227                cov[0] = F::zero();
228                cov.swap(0, c_idx);
229                continue;
230            }
231            active.push(indices[n_active].to_usize().unwrap());
232            n_active += 1;
233
234            if verbose {
235                println!(
236                    "{}\t\t{}\t\t{}\t\t\t\t{}",
237                    n_iter,
238                    active[active.len() - 1],
239                    n_active,
240                    c
241                );
242            }
243        }
244
245        let mut a = l.slice(s![..n_active, ..n_active]).to_owned();
246        let mut b = sign_active.slice(s![..n_active]).to_owned();
247        let mut least_squares = cholesky_solve(&mut a, &mut b);
248
249        let mut aa;
250        if least_squares.len() == 1 && least_squares[0] == F::zero() {
251            least_squares.fill(F::one());
252            aa = F::one();
253        } else {
254            aa = F::one()
255                / (&least_squares * &sign_active.slice(s![..n_active]))
256                    .sum()
257                    .sqrt();
258            if !aa.is_finite() {
259                let mut i = 0;
260                let mut l_ = l.slice_mut(s![..n_active, ..n_active]);
261
262                while !aa.is_finite() {
263                    for j in 0..n_active {
264                        l_[[j, j]] += (F::from(2).unwrap().powi(i)) * eps;
265                    }
266
267                    let mut p = l_.slice(s![.., ..]).to_owned();
268                    let mut q = sign_active.slice(s![..n_active]).to_owned();
269                    least_squares = cholesky_solve(&mut p, &mut q);
270
271                    let tmp = (&least_squares * &sign_active.slice(s![..n_active]))
272                        .sum()
273                        .max(eps);
274                    aa = F::one() / tmp.sqrt();
275
276                    i += 1;
277                }
278            }
279            least_squares *= aa;
280        }
281
282        let corr_eq_dir = gram
283            .slice(s![..n_active, n_active..])
284            .t()
285            .dot(&least_squares);
286
287        let diff = &cov.mapv(|x| c - x);
288        let denom = &corr_eq_dir.mapv(|x| aa - x + tiny_32);
289        let ratio = diff / denom;
290
291        let g1 = match ratio
292            .iter()
293            .cloned()
294            .filter(|&x| x > F::zero())
295            .reduce(F::min)
296        {
297            Some(n) => n,
298            None => F::infinity(),
299        };
300
301        let sum_n = &cov.mapv(|x| c + x);
302        let denom_n = &corr_eq_dir.mapv(|x| aa + x + tiny_32);
303        let ratio_n = sum_n / denom_n;
304
305        let g2 = match ratio_n
306            .iter()
307            .cloned()
308            .filter(|&x| x > F::zero())
309            .reduce(F::min)
310        {
311            Some(n) => n,
312            None => F::infinity(),
313        };
314
315        let gamma_ = g2.min(g1).min(c / aa);
316
317        drop = false;
318
319        let z = &coef.select(Axis(0), &active) * -F::one() / (&least_squares + tiny_32);
320        let z_pos_t = z.iter().cloned().filter(|&x| x > F::zero()).reduce(F::min);
321
322        let z_pos = match z_pos_t {
323            Some(n) => n,
324            None => F::infinity(),
325        };
326
327        if z_pos < gamma_ {
328            let mut idx: Vec<usize> = z
329                .iter()
330                .enumerate()
331                .filter_map(|(i, &value)| if value == z_pos { Some(i) } else { None })
332                .collect();
333
334            idx.reverse();
335
336            for &i in &idx {
337                sign_active[i] = -sign_active[i];
338            }
339
340            drop = true;
341        }
342
343        n_iter += 1;
344
345        if n_iter >= coefs.nrows() {
346            let add_features = 2 * (max_features - n_active).max(1);
347
348            let mut new_coefs = Array2::<F>::zeros((n_iter + add_features, n_features));
349            let old_shape = coefs.nrows();
350            new_coefs.slice_mut(s![..old_shape, ..]).assign(&coefs);
351
352            coefs = new_coefs;
353
354            let mut new_alphas = Array1::<F>::zeros(n_iter + add_features);
355            new_alphas.slice_mut(s![..alphas.len()]).assign(&alphas);
356            alphas = new_alphas;
357        }
358
359        coef.assign(&(coefs.row(n_iter)));
360        prev_coef.assign(&(coefs.row(n_iter - 1)));
361
362        for (i, &idx) in active.iter().enumerate() {
363            coef[idx] = prev_coef[idx] + gamma_ * least_squares[i];
364        }
365
366        coefs.row_mut(n_iter).assign(&coef);
367
368        cov -= &(corr_eq_dir.mapv(|v| gamma_ * v));
369    }
370
371    let alphas_trimmed = alphas.slice(s![..n_iter + 1]).to_owned();
372    let coefs_trimmed = coefs.slice(s![..n_iter + 1, ..]).to_owned();
373    let coefs_t = coefs_trimmed.t().to_owned();
374    (alphas_trimmed, active, coefs_t, n_iter)
375}

377/// Solves a linear system `A * x = b` using a Cholesky factorization.
378///
379/// - When compiled with the `blas` feature:
380///   - Uses LAPACK's Cholesky solver (`potrs`) via `ndarray-linalg`.
381///   - Performs an in-place factorization and solve on the provided arrays.
382/// - Without `blas`:
383///   - Performs a manual two-step triangular solve:  
384///     `L * z = b` then `Lᵀ * x = z`.
385fn cholesky_solve<F: Float>(x: &mut Array2<F>, y: &mut Array1<F>) -> Array1<F> {
386    #[cfg(feature = "blas")]
387    {
388        let mut p_l = x.clone().with_lapack();
389        let p_slice = p_l.as_slice_mut().unwrap();
390        let mut q_l = y.view_mut().with_lapack();
391        let q_slice = q_l.as_slice_mut().unwrap();
392        let shape = x.raw_dim();
393        let layout_m = MatrixLayout::F {
394            col: shape[1] as i32,
395            lda: shape[0] as i32,
396        };
397        F::Lapack::solve_cholesky(layout_m, UPLO::Upper, p_slice, q_slice).unwrap();
398        y.clone()
399    }
400    #[cfg(not(feature = "blas"))]
401    {
402        let mut y_ia = y.view_mut().insert_axis(Axis(1));
403        x.solve_triangular_inplace(&mut y_ia, UPLO::Lower).unwrap();
404        x.t()
405            .solve_triangular_inplace(&mut y_ia, UPLO::Upper)
406            .unwrap();
407        y.clone()
408    }

560    #[test]
561    fn test_covariance() {
562        let dataset = linfa_datasets::diabetes();
563        let x = dataset.records().view();
564        let y = dataset.targets().view();
565        let (_, _, coefs, _) = lars_path(&x, &y, 500, f64::EPSILON, false, 0.0);
566
567        for (index, coef_) in coefs.t().axis_iter(ndarray::Axis(0)).enumerate() {
568            let res = &y - &x.dot(&coef_);
569            let cov = x.t().dot(&res);
570            let ab_cov = cov.abs();
571            let cap_c = ab_cov.max().unwrap();
572            let eps = 1e-3;
573            let threshold = cap_c - eps;
574            let ocur = cov.iter().filter(|v| v.abs() >= threshold).count();
575            if index < x.ncols() {
576                assert!(ocur == index + 1);
577            } else {
578                assert!(ocur == x.ncols());
579            }
580        }
581    }
