24impl<'a, 'b: 'a, F, L, I: TargetDim> Iterator for Iter<'a, 'b, F, L, I> {
25    type Item = (ArrayView1<'a, F>, ArrayView<'a, L, I::Smaller>);
26
27    fn next(&mut self) -> Option<Self::Item> {
28        if self.records.nsamples() <= self.idx {
29            return None;
30        }
31
32        self.idx += 1;
33        let records = self.records.reborrow();
34        let targets = self.targets.clone().reborrow();
35
36        Some((
37            records.index_axis_move(Axis(0), self.idx - 1),
38            targets.index_axis_move(Axis(0), self.idx - 1),
39        ))
40    }
41}

65impl<'a, 'b: 'a, F: 'a, L: 'a, D, T> Iterator for DatasetIter<'a, 'b, ArrayBase<D, Ix2>, T>
66where
67    D: Data<Elem = F>,
68    T: FromTargetArray<'a, Elem = L>,
69{
70    type Item = DatasetView<'a, F, L, T::Ix>;
71
72    fn next(&mut self) -> Option<Self::Item> {
73        if !self.target_or_feature && self.dataset.ntargets() <= self.idx {
74            return None;
75        }
76
77        if self.target_or_feature && self.dataset.nfeatures() <= self.idx {
78            return None;
79        }
80        let mut records = self.dataset.records.view();
81        let mut targets = self.dataset.targets.as_targets();
82        let feature_names;
83        let target_names;
84        let weights = self.dataset.weights.clone();
85
86        if !self.target_or_feature {
87            // This branch should only run for 2D targets
88            targets.collapse_axis(Axis(1), self.idx);
89            feature_names = self.dataset.feature_names.clone();
90            if self.dataset.target_names.is_empty() {
91                target_names = Vec::new();
92            } else {
93                target_names = vec![self.dataset.target_names[self.idx].clone()];
94            }
95        } else {
96            records.collapse_axis(Axis(1), self.idx);
97            target_names = self.dataset.target_names.clone();
98            if self.dataset.feature_names.len() == records.len_of(Axis(1)) {
99                feature_names = vec![self.dataset.feature_names[self.idx].clone()];
100            } else {
101                feature_names = Vec::new();
102            }
103        }
104
105        self.idx += 1;
106
107        let dataset_view = DatasetBase {
108            records,
109            targets,
110            weights,
111            feature_names,
112            target_names,
113        };
114
115        Some(dataset_view)
116    }
117}

147impl<'a, 'b: 'a, F, E: 'b, T> Iterator for ChunksIter<'a, 'b, F, T>
148where
149    T: AsTargets<Elem = E> + FromTargetArray<'b>,
150{
151    type Item = DatasetBase<ArrayView2<'a, F>, T::View>;
152
153    fn next(&mut self) -> Option<Self::Item> {
154        if self.idx == self.records.len_of(self.axis) / self.size {
155            return None;
156        }
157
158        let (mut records, mut targets) = (
159            self.records.reborrow(),
160            self.targets.as_targets().reborrow(),
161        );
162        records.slice_axis_inplace(
163            self.axis,
164            (self.idx * self.size..(self.idx + 1) * self.size).into(),
165        );
166        targets.slice_axis_inplace(
167            self.axis,
168            (self.idx * self.size..(self.idx + 1) * self.size).into(),
169        );
170
171        self.idx += 1;
172
173        Some(DatasetBase::new(records, T::new_targets_view(targets)))
174    }
