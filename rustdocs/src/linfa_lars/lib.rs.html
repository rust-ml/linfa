1//! # Least angle regression a.k.a. LAR
2//!
3//! This struct contains the parameters of a fitted LARS model. This includes the seperating
4//! hyperplane, (optionally) intercept, alphas (Maximum of covariances (in absolute value) at each iteration),
5//! Indices of active variables at the end of the path,
6//!
7//! LARS is similar to forward stepwise regression.
8//! At each step, it finds the feature most correlated with the target.
9//! When there are multiple features having equal correlation, instead of continuing along the same feature,
10//! it proceeds in a direction equiangular between the features.
11//!
12//! ## References
13//!
14//! * ["Least Angle Regression", Efron et al.](https://web.stanford.edu/~hastie/Papers/LARS/LeastAngle_2002.pdf)
15//! * [Wikipedia entry on the Least-angle regression](https://en.wikipedia.org/wiki/Least-angle_regression)
16//! * [Scikit-Learn User Guide](https://scikit-learn.org/stable/modules/linear_model.html#least-angle-regression)
17//!
18//! # Examples
19//! ```
20//! use linfa::prelude::*;
21//! use linfa_lars::Lars;
22//!
23//! // Load the Diabetes dataset
24//! let dataset = linfa_datasets::diabetes();
25//!
26//! // Create a LARS model with intercept fitting enabled
27//! let lars = Lars::params().fit_intercept(true).fit(&dataset).unwrap();
28//!
29//! // Generate predictions using the trained model and compute the RÂ² score of the predictions
30//! let pred = lars.predict(&dataset);
31//! let r2 = pred.r2(&dataset).unwrap();
32//! println!("r2 from prediction: {}", r2);
33//! ```

50#[derive(Debug, Clone)]
51pub struct Lars<F> {
52    hyperplane: Array1<F>,
53    intercept: F,
54    alphas: Array1<F>,
55    n_iter: usize,
56    active: Vec<usize>,
57    coef_path: Array2<F>,
58}
59
60impl<F: Float> Lars<F> {
61    /// Create default Lars hyper parameters
62    ///
63    /// By default, an intercept will be fitted. To disable fitting an
64    /// intercept, call `.fit_intercept(false)` before calling `.fit()`.
65    ///
66    /// The feature matrix will not be normalized by default.
67    pub fn params() -> LarsParams<F> {
68        LarsParams::new()
69    }
70
71    /// Get the varying values of the coefficients along the path.
72    pub fn coef_path(&self) -> &Array2<F> {
73        &self.coef_path
74    }
75
76    /// Get the fitted hyperplane
77    pub fn hyperplane(&self) -> &Array1<F> {
78        &self.hyperplane
79    }
80
81    /// Maximum of covariances (in absolute value) at each iteration
82    pub fn alphas(&self) -> &Array1<F> {
83        &self.alphas
84    }
85
86    /// The number of iterations taken by the algorithm to find the grid of alphas for each target
87    pub fn n_iter(&self) -> usize {
88        self.n_iter
89    }
90
91    /// Indices of active variables at the end of the path
92    pub fn active(&self) -> &Vec<usize> {
93        &self.active
94    }
95
96    /// Get the fitted intercept, 0. if no intercept was fitted
97    pub fn intercept(&self) -> F {
98        self.intercept
99    }
100}
101
102impl<F: Float, D: Data<Elem = F>> PredictInplace<ArrayBase<D, Ix2>, Array1<F>> for Lars<F> {
103    /// Given an input matrix `X`, with shape `(n_samples, n_features)`,
104    /// `predict` returns the target variable according to LARS
105    /// learned from the training data distribution.
106    fn predict_inplace<'a>(&'a self, x: &'a ArrayBase<D, Ix2>, y: &mut Array1<F>) {
107        assert_eq!(
108            x.nrows(),
109            y.len(),
110            "The number of data points must match the number of output targets."
111        );
112
113        *y = x.dot(&self.hyperplane) + self.intercept;
114    }
115    fn default_target(&self, x: &ArrayBase<D, Ix2>) -> Array1<F> {
116        Array1::zeros(x.nrows())
117    }
118}
