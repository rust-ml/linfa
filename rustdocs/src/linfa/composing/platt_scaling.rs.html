1//! Implement Platt calibration with Newton method
2//!
3//! Platt scaling is a way of transforming the outputs of a classification model into a probability
4//! distribution over classes. It is, for example, used in the calibration of SVM models to predict
5//! plausible probability values.
6//!
7//! # Example
8//!
9//! ```rust, ignore
10//! let model = ...;
11//!
12//! let model = Platt::params()
13//!      .fit_with(model, &train)?;
14//!
15//! let pred: Array1<Pr> = model.predict(&valid);
16//! ```
17

29/// Fitted Platt model
30///
31/// This model contains a sigmoid scaling parameters and a second univariate, uncalibrated model.
32/// The output of the uncalibrated model is scaled with the following function:
33/// ```text
34/// g(x) = 1 / (1 + exp(A * f(x) + B)
35/// ```
36///
37/// The scaling factors `A` and `B` are estimated with the Newton's method, presented in the
38/// following paper: <https://www.csie.ntu.edu.tw/~cjlin/papers/plattprob.pdf>
39#[derive(Debug, Clone, PartialEq, Eq)]
40pub struct Platt<F, O> {
41    a: F,
42    b: F,
43    obj: O,
44}
45
46/// Parameters for Platt's Newton method
47#[derive(Debug, Clone, PartialEq, Eq)]
48pub struct PlattValidParams<F, O> {
49    maxiter: usize,
50    minstep: F,
51    sigma: F,
52    phantom: PhantomData<O>,
53}
54
55#[derive(Debug, Clone, PartialEq, Eq)]
56pub struct PlattParams<F, O>(PlattValidParams<F, O>);
57
58impl<F: Float, O> Default for PlattParams<F, O> {
59    fn default() -> Self {
60        Self(PlattValidParams {
61            maxiter: 100,
62            minstep: F::cast(1e-10),
63            sigma: F::cast(1e-12),
64            phantom: PhantomData,
65        })
66    }
67}
68
69impl<F: Float, O> PlattParams<F, O> {
70    /// Set the maximum number of iterations in the optimization process
71    ///
72    /// The Newton's method is an iterative optimization process, which uses the first and second
73    /// order gradients to find optimal `A` and `B`. This function caps the maximal number of
74    /// iterations.
75    pub fn maxiter(mut self, maxiter: usize) -> Self {
76        self.0.maxiter = maxiter;
77
78        self
79    }
80
81    /// Set the minimum stepsize in the line search
82    ///
83    /// After estimating the Hessian matrix, a line search is performed to find the optimal step
84    /// size in each optimization step. In each attempt the stepsize is halfed until this threshold
85    /// is reached. After reaching the threshold the algorithm fails because the desired precision
86    /// could not be achieved.
87    pub fn minstep(mut self, minstep: F) -> Self {
88        self.0.minstep = minstep;
89
90        self
91    }
92
93    /// Set the Hessian's sigma value
94    ///
95    /// The Hessian matrix is regularized with H' = H + sigma I to avoid numerical issues. This
96    /// function set the amount of regularization.
97    pub fn sigma(mut self, sigma: F) -> Self {
98        self.0.sigma = sigma;
99
100        self
101    }
102}

104impl<F: Float, O> ParamGuard for PlattParams<F, O> {
105    type Checked = PlattValidParams<F, O>;
106    type Error = PlattError;
107
108    fn check_ref(&self) -> Result<&Self::Checked, PlattError> {
109        if self.0.maxiter == 0 {
110            Err(PlattError::MaxIterReached)
111        } else if self.0.minstep.is_negative() {
112            Err(PlattError::MinStepNegative(
113                self.0.minstep.to_f32().unwrap(),
114            ))
115        } else if self.0.sigma.is_negative() {
116            Err(PlattError::SigmaNegative(self.0.sigma.to_f32().unwrap()))
117        } else {
118            Ok(&self.0)
119        }
120    }
121
122    fn check(self) -> Result<Self::Checked, PlattError> {
123        self.check_ref()?;
124        Ok(self.0)
125    }
126}

134/// Platt Newton's method errors
135///
136/// Errors occur when setting invalid parameters or the optimization process fails.
137pub enum PlattError {
138    #[error("line search did not converge")]
139    LineSearchNotConverged,
140    #[error("platt scaling did not converge")]
141    MaxIterReached,
142    #[error("maxiter should be larger than zero")]
143    MaxIterZero,
144    #[error("minstep should be positive, is {0}")]
145    MinStepNegative(f32),
146    #[error("sigma should be positive, is {0}")]
147    SigmaNegative(f32),
148    #[error(transparent)]
149    LinfaError(#[from] crate::error::Error),
150}

153    /// Create default parameter set for the Platt scaling algorithm
154    ///
155    /// The default values are:
156    /// * `maxiter`: 100,
157    /// * `minstep`: 1e-10,
158    /// * `sigma`: 1e-12
159    ///
160    pub fn params() -> PlattParams<F, O> {
161        PlattParams::default()
162    }

165impl<'a, F: Float, O: 'a> FitWith<'a, Array2<F>, Array1<bool>, PlattError>
166    for PlattValidParams<F, O>
167where
168    O: PredictInplace<Array2<F>, Array1<F>>,
169{
170    type ObjectIn = O;
171    type ObjectOut = Platt<F, O>;
172
173    /// Calibrate another model with Platt scaling
174    ///
175    /// This function takes another model and binary decision dataset and calibrates it to produce
176    /// probability values. The returned model therefore implements the prediction trait for
177    /// probability targets.
178    fn fit_with(
179        &self,
180        obj: O,
181        ds: &DatasetBase<Array2<F>, Array1<bool>>,
182    ) -> Result<Self::ObjectOut, PlattError> {
183        let predicted = obj.predict(ds);
184
185        let (a, b) = platt_newton_method(predicted.view(), ds.targets().view(), self)?;
186
187        Ok(Platt { a, b, obj })
188    }
189}

191impl<F: Float, D, O> PredictInplace<ArrayBase<D, Ix2>, Array1<Pr>> for Platt<F, O>
192where
193    D: Data<Elem = F>,
194    O: PredictInplace<ArrayBase<D, Ix2>, ArrayBase<D, Ix1>>,
195{
196    fn predict_inplace(&self, data: &ArrayBase<D, Ix2>, targets: &mut Array1<Pr>) {
197        assert_eq!(
198            data.nrows(),
199            targets.len(),
200            "The number of data points must match the number of output targets."
201        );
202        for (x, target) in self.obj.predict(data).iter().zip(targets.iter_mut()) {
203            *target = platt_predict(*x, self.a, self.b);
204        }
205    }
206
207    fn default_target(&self, x: &ArrayBase<D, Ix2>) -> Array1<Pr> {
208        Array1::default(x.nrows())
209    }
210}

212/// Predict a probability with the sigmoid function
213///
214/// Similar to stable sigmoid implementations
215/// <https://stackoverflow.com/questions/51976461/optimal-way-of-defining-a-numerically-stable-sigmoid-function-for-a-list-in-pyth>
216///
217/// # Parameters
218/// * `x`: uncalibrated value f(x)
219/// * `a`: parameter A,
220/// * `b`: parameter B,
221pub fn platt_predict<F: Float>(x: F, a: F, b: F) -> Pr {
222    let f_apb = a * x + b;
223    let f_apb = f_apb.to_f32().unwrap();
224
225    // avoid numerical problems for large f_apb
226    if f_apb >= 0.0 {
227        Pr::new((-f_apb).exp() / (1.0 + (-f_apb).exp()))
228    } else {
229        Pr::new(1.0 / (1.0 + f_apb.exp()))
230    }
231}

233/// Run Newton's method to find optimal `A` and `B` values
234///
235/// The optimization process happens in two steps, first the closed-form Hessian matrix and
236/// gradient vectors are calculated. Then a line-search tries to find the optimal learning rate
237/// for each step.
238//#[allow(clippy::suspicious_operation_groupings)]
239pub fn platt_newton_method<'a, F: Float, O>(
240    reg_values: ArrayView1<'a, F>,
241    labels: ArrayView1<'a, bool>,
242    params: &PlattValidParams<F, O>,
243) -> Result<(F, F), PlattError> {
244    let (num_pos, num_neg) = labels.iter().fold((0, 0), |mut val, x| {
245        match x {
246            true => val.0 += 1,
247            false => val.1 += 1,
248        }
249
250        val
251    });
252    let (num_pos, num_neg) = (num_pos as f32, num_neg as f32);
253
254    let (hi_target, lo_target) = ((num_pos + 1.0) / (num_pos + 2.0), 1.0 / (num_neg + 2.0));
255
256    let target_values = labels
257        .iter()
258        .map(|x| if *x { hi_target } else { lo_target })
259        .map(|x| F::from(x).unwrap())
260        .collect::<Vec<_>>();
261
262    let reg_values = reg_values
263        .into_iter()
264        .map(|x| F::from(*x).unwrap())
265        .collect::<Vec<_>>();
266
267    let mut a = F::zero();
268    let mut b = F::from((num_neg + 1.0) / (num_pos + 1.0)).unwrap().ln();
269    let mut fval = F::zero();
270
271    for (v, t) in reg_values.iter().zip(target_values.iter()) {
272        let f_apb = *v * a + b;
273        if f_apb >= F::zero() {
274            fval += *t * f_apb + (F::one() + (-f_apb).exp()).ln();
275        } else {
276            fval += (*t - F::one()) * f_apb + (F::one() + f_apb.exp()).ln();
277        }
278    }
279
280    let mut idx = 0;
281    for _ in 0..params.maxiter {
282        let (mut h11, mut h22) = (params.sigma, params.sigma);
283        let (mut h21, mut g1, mut g2) = (F::zero(), F::zero(), F::zero());
284
285        for (v, t) in reg_values.iter().zip(target_values.iter()) {
286            let f_apb = *v * a + b;
287
288            let (p, q) = if f_apb >= F::zero() {
289                (
290                    (-f_apb).exp() / (F::one() + (-f_apb).exp()),
291                    F::one() / (F::one() + (-f_apb).exp()),
292                )
293            } else {
294                (
295                    F::one() / (F::one() + f_apb.exp()),
296                    f_apb.exp() / (F::one() + f_apb.exp()),
297                )
298            };
299
300            let d2 = p * q;
301            h11 += *v * *v * d2;
302            h22 += d2;
303            h21 += *v * d2;
304
305            let d1 = *t - p;
306            g1 += *v * d1;
307            g2 += d1;
308        }
309
310        if g1.abs() < F::from(1e-5 * reg_values.len() as f32).unwrap()
311            && g2.abs() < F::from(1e-5 * reg_values.len() as f32).unwrap()
312        {
313            break;
314        }
315
316        let det = h11 * h22 - h21.powi(2);
317        let d_a = -(h22 * g1 - h21 * g2) / det;
318        let d_b = -(-h21 * g1 + h11 * g2) / det;
319        let gd = g1 * d_a + g2 * d_b;
320
321        let mut stepsize = F::one();
322        while stepsize >= params.minstep {
323            let new_a = a + stepsize * d_a;
324            let new_b = b + stepsize * d_b;
325            let mut newf = F::zero();
326
327            for (v, t) in reg_values.iter().zip(target_values.iter()) {
328                let f_apb = *v * new_a + new_b;
329
330                if f_apb >= F::zero() {
331                    newf += *t * f_apb + (F::one() + (-f_apb).exp()).ln();
332                } else {
333                    newf += (*t - F::one()) * f_apb + (F::one() + f_apb.exp()).ln();
334                }
335            }
336
337            if newf < fval + F::from(1e-4).unwrap() * stepsize * gd {
338                a = new_a;
339                b = new_b;
340                fval = newf;
341                break;
342            } else {
343                stepsize /= F::one() + F::one();
344            }
345
346            if stepsize < params.minstep {
347                return Err(PlattError::LineSearchNotConverged);
348            }
349        }
350
351        idx += 1;
352    }
353
354    if params.maxiter == idx {
355        return Err(PlattError::MaxIterReached);
356    }
357
358    Ok((a, b))
359}

373    /// Generate dummy values which can be predicted with the Platt model
374    fn generate_dummy_values<F: Float, R: Rng>(
375        a: F,
376        b: F,
377        n: usize,
378        rng: &mut R,
379    ) -> (Array1<F>, Array1<bool>) {
380        // generate probability values, omit p = 0.0, p = 1.0 to avoid infinity in reverse function
381        let prob_values = Array1::linspace(
382            F::one() / F::from(n).unwrap(),
383            F::one() - F::one() / F::from(n).unwrap(),
384            n - 2,
385        );
386
387        // generate regression values with inverse function
388        let reg_values = prob_values
389            .iter()
390            .map(|x| (F::one() - *x) / *x)
391            .map(|x| (x.ln() - b) / a)
392            .collect();
393
394        // roll decision according to probability
395        let decisions = prob_values
396            .iter()
397            .map(|x| rng.gen_bool(x.to_f64().unwrap()))
398            .collect();
399
400        (reg_values, decisions)
401    }
